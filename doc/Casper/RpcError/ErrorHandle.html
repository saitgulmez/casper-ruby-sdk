def self.error_handling(url)
  begin
    response = RestClient.get(url)
    # response # {"jsonrpc":"2.0","id":null,"error":{"code":-32700,"message":"Parse error","data":null}}
    parsed = JSON.parse(response) # {"jsonrpc"=>"2.0", "id"=>nil, "error"=>{"code"=>-32700, "message"=>"Parse error", "data"=>nil}}
    parsed
    # # parsed["jsonrpc"]
    # # parsed["error"]
    # # parsed["error"]["message"]
    # # parsed["error"]["data"]
    # if parsed["error"]["data"] == nil
    #   parsed["error"]["message"]
    # end
  rescue RestClient::ResourceNotFound => e
    p e.class
    # "ResourceNotFound"
  rescue Errno::ECONNREFUSED => e
    p e.class
    # "Errno::ECONNREFUSED"
  rescue SocketError => e
    p e.class
    # "Socket Error"
  end
end
